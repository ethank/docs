## Tech Stack

### Infrastructure
- **Neon** - Postgres with pgvector (preset embeddings)
- **Railway** - API + workers deployment
- **Clerk** - Authentication (social login, user management)
- **Portkey** - LLM gateway (OpenAI + Anthropic fallbacks)
- **Sentry** - Error tracking
- **PostHog** - Product analytics
- **Vercel** - Frontend (Next.js)

### AI/ML
- **Primary**: GPT-4o-mini (fast, cheap for conversations)
- **Fallback**: Claude 3.5 Sonnet (better reasoning if needed)
- **Future**: GPT-4o for audio analysis (when we add uploads)

**Cost:** Around 50-100 per month infrastructure + usage-based LLM costs

```typescript src/middleware/auth.ts
import { clerkClient } from '@clerk/clerk-sdk-node';
import { Request, Response, NextFunction } from 'express';

export async function authenticate(
  req: Request,
  res: Response,
  next: NextFunction
) {
  const token = req.headers.authorization?.substring(7);

  if (!token) {
    return res.status(401).json({ error: 'Unauthorized' });
  }

  try {
    const decoded = await clerkClient.verifyToken(token);
    const user = await db.getUserByClerkId(decoded.sub);

    req.user = user;
    next();
  } catch (error) {
    return res.status(401).json({ error: 'Invalid token' });
  }
}
```

```typescript src/services/llm.ts
  const response = await portkey.chat.completions.create({
    messages: params.messages,
    tools: params.tools,
    max_tokens: params.maxTokens,
    temperature: params.temperature

  await logLLMCall({
    model: response.model,
    provider: response._headers['x-portkey-provider'],
    usage: response.usage,
    latencyMs,
    status: 'success'
  });

  return response;
}
```

### 2.4 Vector Search Implementation

```typescript
// src/services/tools.ts

async function search_presets(args: {
  description: string;
  tags?: string[];
  pluginId?: string;
  limit?: number;
}) {
  const { description, tags, pluginId, limit = 5 } = args;

  // Generate embedding for search query
  const embedding = await generateEmbedding(description);

  // Vector similarity search with filters
  const results = await sql`
    WITH ranked_presets AS (
      SELECT
        p.*,
        pl.name as plugin_name,
        pl.is_uad,
        pl.price_usd,
        pl.purchase_url,
        1 - (p.embedding <=> ${embedding}::vector) as similarity,
        CASE WHEN p.plugin_id = ANY(${userPlugins})
          THEN 1 ELSE 0 END as owned
      FROM presets p
      JOIN plugins pl ON pl.id = p.plugin_id
      WHERE
        p.embedding IS NOT NULL
        ${pluginId ? sql`AND p.plugin_id = ${pluginId}` : sql``}
        ${tags ? sql`AND p.tags && ${tags}` : sql``}
      ORDER BY p.embedding <=> ${embedding}::vector
      LIMIT ${limit * 2}
    )
    SELECT * FROM ranked_presets
    WHERE similarity > 0.7
    ORDER BY owned DESC, similarity DESC
    LIMIT ${limit}
  `;

  return {
    presets: results.map(r => ({
      id: r.id,
      name: r.preset_name,
      description: r.description,
      plugin: r.plugin_name,
      tags: r.tags,
      characteristics: r.characteristics,
      similarity: r.similarity,
      owned: r.owned,
      price: r.price_usd,
      purchaseUrl: r.purchase_url
    }))
  };
}
```

```typescript components/ChatInterface.tsx
import { useState } from 'react';

export default function ChatInterface() {
  const { getToken } = useAuth();
  const [messages, setMessages] = useState([]);
  const [input, setInput] = useState('');

  async function sendMessage() {
    const token = await getToken();

    const response = await fetch('/api/chat', {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${token}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({ message: input })
    });

    const data = await response.json();
    setMessages([...messages, data]);
  }

  return (
    <div className="flex flex-col h-screen">
      <div className="flex-1 overflow-y-auto p-4">
        {messages.map((msg, i) => (
          <div key={i}>{msg.content}</div>
        ))}
      </div>
      <div className="p-4 border-t">
        <input
          value={input}
          onChange={(e) => setInput(e.target.value)}
          onKeyPress={(e) => e.key === 'Enter' && sendMessage()}
          className="w-full px-4 py-2 border rounded"
          placeholder="Describe the sound you want..."
        />
      </div>
    </div>
  );
}
```

### 4.1 Generate Embeddings

```typescript
// scripts/generate-embeddings.ts

import { neon } from '@neondatabase/serverless';
import OpenAI from 'openai';

const sql = neon(process.env.DATABASE_URL!);
const openai = new OpenAI();

async function generateEmbeddings() {
  const presets = await sql`
    SELECT id, preset_name, description, tags, characteristics
    FROM presets WHERE embedding IS NULL
  `;

  console.log(`Generating embeddings for ${presets.length} presets...`);

  for (const preset of presets) {
    const text = `
      Preset: ${preset.preset_name}
      Description: ${preset.description}
      Tags: ${preset.tags.join(', ')}
      Characteristics: ${JSON.stringify(preset.characteristics)}
    `.trim();

    const response = await openai.embeddings.create({
      model: 'text-embedding-ada-002',
      input: text
    });

    const embedding = response.data[0].embedding;

    await sql`
      UPDATE presets
      SET embedding = ${JSON.stringify(embedding)}::vector
      WHERE id = ${preset.id}
    `;

    console.log(`✓ ${preset.preset_name}`);
  }

  console.log('Done!');
}

generateEmbeddings();
```

```sql
SELECT DATE(created_at), COUNT(DISTINCT user_id)
FROM sessions
WHERE created_at > NOW() - INTERVAL '7 days'
GROUP BY DATE(created_at);

-- Preset performance
SELECT
  p.preset_name,
  p.plugin_id,
  COUNT(DISTINCT ua.user_id) as views,
  AVG((ua.metadata->>'helpful')::float) as helpfulness
FROM presets p
LEFT JOIN user_actions ua ON ua.target_id = p.id::text
GROUP BY p.id, p.preset_name, p.plugin_id
ORDER BY views DESC LIMIT 20;
```

- Day 7 retention more than 25%
- NPS more than 30
- Clear user persona
- Sustainable unit economics
- Team excited to maintain it

**If YES → Fill:**
- Allocate ongoing resources (0.25-0.5 FTE)
- Announce publicly
- Set 90-day roadmap
- Move from "experiment" to "product"
  </Tab>

  <Tab title="Kill Criteria">
**Kill if Missing Any:**
- Less than 25% Day 7 retention
- NPS less than 30
- No clear use cases
- Cost per session more than 0.50
- Team not excited to maintain

**If NO → Kill:**
- Document learnings (30-60 min)
- Archive code
- Thank beta users, offer alternative
- Move team to next experiment
- No shame - killing fast is the goal
  </Tab>
</Tabs>
